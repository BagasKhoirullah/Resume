Named Constructor

class Mobil {

String? merk;
int? tahun;

Mobil(this.merk,this.tahun);

Mobil.baru(){
merk='Toyota';
tahun=2023;
}

Mobil.dariJson(Map json){
merk=json['merk'];
tahun=json['tahun'];
}

}
