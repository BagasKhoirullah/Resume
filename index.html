Named Constructor

class Mobil {

String? merk;
int? tahun;

Mobil(this.merk, this.tahun);

Mobil.baru(){
merk = 'Toyota';
tahun = 2023;
}

Mobil.dariJson(Map json){
merk = json['merk'];
tahun = json['tahun'];
}

}

Constant Constructor

class Warna {

final int r,g,b;

const Warna(this.r,this.g,this.b);

}

Enkapsulasi

Enkapsulasi adalah konsep menyembunyikan detail implementasi dan hanya memberikan akses melalui method tertentu.

  • Keamanan data
  • Kontrol akses
  • Menggunakan _ untuk private

Getter

class Person {

String _name;

String get nama {
return _name;
}

}

Setter

set nama(String value){

if(value.length >= 3){
_name = value;
}

}

Studi Kasus Login

class User {

String _username;
String _password;

User(this._username,this._password);

bool login(String u,String p){
return _username == u && _password == p;
}

}
