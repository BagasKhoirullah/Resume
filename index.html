Review OOP

  • OOP adalah paradigma berbasis objek
  • Class adalah blueprint
  • Object adalah instance
  • Constructor untuk inisialisasi objek
  • Contoh cepat:
class Buku {
  String judul;
  Buku(this.judul);
  
}

Jenis Constructor

  • Default Constructor: ClassName( )
  • Named Constructor: ClassName.nama( )
  • Parameterized Constructor: ClassName(this.param)
  • Constant Constructor: const ClassName( )
  • Factory Constructor: (akan dibahas di sesi lanjut)

Named Constructor

class Mobil {
  String? merk;
  int? tahun;
  
// Parameterized constructor
Mobil(this.merk, this.tahun);
  
// Named constructor
Mobil.baru() {
  merk = 'Toyota';
  tahun = 2023;
}
  
// Named constructor dengan parameter
Mobil.dariJson(Map json) {
  merk = json['merk'];
  tahun = json['tahun'];
}
  
// Override toString for better print output
@override
String toString() {
  return 'Mobil{merk: $merk, tahun: $tahun}';
  }
}

Constant Constructor

  • Digunakan untuk objek yang immutable (tidak berubah)
  • Semua field harus final
  • Menghemat memory untuk objek yang sama
class Warna {
  final int r, g, b;
  const Warna(this.r, this.g, this.b);
  @override
  String toString() {
  return 'Warna(r: $r, g: $g, b: $b)';
  }
}
void main() {
  const merah = Warna(255, 0, 0);
  const merahLagi = Warna(255, 0, 0); // Menggunakan instance yang sama di memory
  print(merah);
  print(merahLagi);
}

Pengantar Enkapsulasi

  • Konsep: Menyembunyikan detail implementasi dari luar.
  • Tujuan: Keamanan data, Kontrol akses, Fleksibilitas perubahan
  • Cara di Dart: Menggunakan underscore (_ ) untuk private members.

Public vs Private

  • Public: Dapat diakses dari mana saja (default).
  • Private: Hanya dapat diakses dalam file yang sama (gunakan _ ).
class Person {
  String _name;
  int age;
  Person(this._name,this.age);
  
}

Getter

  • Getter memungkinkan membaca data private dengan kontrol.
  • Sintaks 1: tipeData get namaGetter => _field;
  • Sintaks 2: tipeData namaGetter() { _field; }
class Person {
  String _name;
  int age;
  Person(this._name, this.age);
  
  String get getName1 {
  return _name;
  }
  
  String getName() {
  return _name;
  }
}

Setter

  • Setter memungkinkan modifikasi data dengan validasi.
  • Sintaks 1: set namaSetter(value) { ... }
  • Sintaks 2: void namaSetter(value) { ... }
class Person {
  String _name;
  int age;
  Person(this._name, this.age);
  
  set setName1(String name) {
    _name = name;
  }
  
  void setName(String name) {
    _name = name;
  }
  
  String getName() {
    return _name;
  }
}

Getter & Setter dengan Validasi

class Produk {
  String _nama;
  double _harga;
  Produk(this._nama, this._harga);
  // Getter
  String get nama => _nama;
  double get harga => _harga;
  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('Gagal memperbarui nama: Nama harus memiliki minimal 3 karakter.');
    }
  }

  // Setter untuk harga (dipindahkan ke dalam kelas Produk)
  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('Gagal memperbarui harga: Harga harus lebih besar dari 0.');
    }
  }
}

class User {

  String _username;
  String _password;
  String _email;

  User(this._username, this._password, this._email);

  // Getter
  String get username => _username;
  String get password => _password;
  String get email => _email;

  // Setter Username
  set username(String value) {
    if (value.length >= 5) {
      _username = value;
      print("Username berhasil diubah menjadi: $_username");
    } else {
      print("Username gagal diubah. Minimal 5 karakter.");
    }
  }

  // Setter Password
  set password(String value) {
    if (value.length >= 8) {
      _password = value;
      print("Password berhasil diubah.");
    } else {
      print("Password gagal diubah. Minimal 8 karakter.");
    }
  }

  // Setter Email
  set email(String value) {
    if (value.contains("@")) {
      _email = value;
      print("Email berhasil diubah menjadi: $_email");
    } else {
      print("Email tidak valid. Harus mengandung '@'.");
    }
  }
}

void main() {

  // Membuat objek
  User user1 = User("bagas123", "password123", "bagas@email.com");

  // Menampilkan data awal
  print("Username: ${user1.username}");
  print("Password: ${user1.password}");
  print("Email: ${user1.email}");

  print("");

  // Mencoba mengubah data
  user1.username = "bag";        // gagal
  user1.username = "bagas_k";    // berhasil

  user1.password = "123";        // gagal
  user1.password = "secure123";  // berhasil

  user1.email = "bagasemail";    // gagal
  user1.email = "bagas@mail.com";// berhasil

}

class User {
  String _username;
  String _password;
  User(this._username, this._password);
  String get username => _username;
  bool login(String username, String password) {
    return _username == username && _password == password;
  }
  
  void gantiPassword(String lama, String baru) {
    if (_password == lama) { // Now _password refers to the instance variable of this User object
      if (baru.length >= 8) {
        _password = baru;
        print('Password berhasil diubah.');
      } else {
        print('Gagal mengubah password: Password baru harus memiliki minimal 8 karakter.');
      }
    } else {
      print('Gagal mengubah password: Password lama tidak sesuai.');
    }
  }
}

class RekeningBank {

  String _nomorRekening;
  double _saldo;

  // Constructor
  RekeningBank(this._nomorRekening, this._saldo);

  // Method untuk menambah saldo
  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print("Deposit berhasil: Rp$jumlah");
    } else {
      print("Jumlah deposit tidak valid.");
    }
  }

  // Method untuk menarik saldo
  void tarik(double jumlah) {
    if (jumlah <= 0) {
      print("Jumlah penarikan tidak valid.");
    } else if (jumlah > _saldo) {
      print("Saldo tidak mencukupi.");
    } else {
      _saldo -= jumlah;
      print("Penarikan berhasil: Rp$jumlah");
    }
  }

  // Method untuk mengecek saldo
  void cekSaldo() {
    print("Nomor Rekening : $_nomorRekening");
    print("Saldo Saat Ini : Rp$_saldo");
  }
}

void main() {

  // Membuat objek rekening
  RekeningBank rekening1 = RekeningBank("251410016", 900000);

  print("=== Saldo Awal ===");
  rekening1.cekSaldo();

  print("\n=== Deposit ===");
  rekening1.deposit(200000);

  print("\n=== Tarik Saldo ===");
  rekening1.tarik(100000);

  print("\n=== Coba Tarik Melebihi Saldo ===");
  rekening1.tarik(1000000);

  print("\n=== Saldo Akhir ===");
  rekening1.cekSaldo();

}
