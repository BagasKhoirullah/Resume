Named Constructor

class Mobil {

String? merk;
int? tahun;

Mobil(this.merk,this.tahun);

Mobil.baru(){

merk='Toyota';
tahun=2023;

}

Mobil.dariJson(Map json){

merk=json['merk'];
tahun=json['tahun'];

}

}

Praktik 1 – Multiple Constructor

class Mahasiswa {

String? nama;
int? umur;

Mahasiswa(this.nama,this.umur);

Mahasiswa.dariMap(Map data){

nama=data['nama'];
umur=data['umur'];

}

Mahasiswa.tanpaNama(){

nama="Tanpa Nama";
umur=0;

}

}

Pengantar Enkapsulasi

  • Menyembunyikan detail implementasi
  • Melindungi data
  • Kontrol akses

Public vs Private

class Person {

String _name;
int age;

Person(this._name,this.age);

}

Getter

class Person {

String _name;
int age;

Person(this._name,this.age);

String get getName => _name;

}

Setter

class Person {

String _name;

Person(this._name);

set setName(String name){

_name=name;

}

}

Getter & Setter dengan Validasi

class Produk {

String _nama;
double _harga;

Produk(this._nama,this._harga);

String get nama => _nama;

set nama(String value){

if(value.length>=3){
_nama=value;
}

}

}

Studi Kasus Login

class User {

String _username;
String _password;

User(this._username,this._password);

bool login(String username,String password){

return username==_username &&
password==_password;

}

}
