class Buku {

Jenis Constructor

  • Default Constructor
  • Named Constructor
  • Parameterized Constructor
  • Constant Constructor
  • Factory Constructor

Named Constructor

class Mobil {

String? merk;
int? tahun;

Mobil(this.merk,this.tahun);

Mobil.baru(){
merk='Toyota';
tahun=2023;
}

Mobil.dariJson(Map json){
merk=json['merk'];
tahun=json['tahun'];
}

}

Constant Constructor

Digunakan untuk objek immutable. Semua field harus final.

class Warna{

final int r,g,b;

const Warna(this.r,this.g,this.b);

}

print("Titik dibuat");

Praktik 1 – Multiple Constructor

Buat kelas Mahasiswa dengan beberapa constructor.

class Mahasiswa{

String? nama;
int? umur;

Mahasiswa(this.nama,this.umur);

Mahasiswa.dariMap(Map data){
nama=data['nama'];
umur=data['umur'];
}

Mahasiswa.tanpaNama(){
nama="Tanpa Nama";
umur=0;
}

}

Pengantar Enkapsulasi

  • Menyembunyikan detail implementasi
  • Melindungi data
  • Mengontrol akses
  • Menggunakan underscore (_) untuk private

Getter

class Person{

String _name;

String get nama => _name;

}

Setter

set nama(String value){

_name=value;

}

Getter & Setter dengan Validasi

class Produk{

String _nama;
double _harga;

Produk(this._nama,this._harga);

String get nama => _nama;
double get harga => _harga;

set harga(double value){

if(value>0){
_harga=value;
}

}

}

Studi Kasus Login

class User{

String _username;
String _password;

User(this._username,this._password);

bool login(String u,String p){

return _username==u && _password==p;

}

}
