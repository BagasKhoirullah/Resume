Named Constructor

class Mobil {

String? merk;
int? tahun;

Mobil(this.merk,this.tahun);

Mobil.baru(){
merk='Toyota';
tahun=2023;
}

Mobil.dariJson(Map json){
merk=json['merk'];
tahun=json['tahun'];
}

}

Constant Constructor

Digunakan untuk objek immutable.

class Warna {

final int r,g,b;

const Warna(this.r,this.g,this.b);

}

Praktik 1 – Multiple Constructor

Buat kelas Mahasiswa dengan beberapa constructor.

class Mahasiswa {

String? nama;
int? umur;

Mahasiswa(this.nama,this.umur);

Mahasiswa.dariMap(Map data){
nama=data['nama'];
umur=data['umur'];
}

Mahasiswa.tanpaNama(){
nama="Tanpa Nama";
umur=0;
}

}

Studi Kasus Login

class User {

String _username;
String _password;

User(this._username,this._password);

bool login(String username,String password){
return _username==username && _password==password;
}

}
